mpl.rcParams['font.size'] = 16.0    # globally assign fontsize for matplotlib text

fig = plt.figure(figsize=[16, 6])  # initialte figure object
ax = fig.add_subplot(1,2,1)         # add one axes in figure

ax.hist(
    [df.loc[lambda x: x.Sex=='female', 'Age'], 
     df.loc[lambda x: x.Sex=='male', 'Age']], # create two datasets to visual
    bins=18,       # how many bins
    range=[0, 90], # under the range to consider
    normed=False,  # whether normalize by integration = 1
    weights=None,  # if points have different importance
    cumulative=False, # aggregative vs. distributional
    histtype='bar',  # may change to barstacked but not necessary
    stacked=True,    # whether stacked for multiple data source or not
    align='mid',     # usually align to the middle of the bin
    orientation='vertical',  # usually use vertical
    log=False, # if the scale are dramatically different across the bin
    color=['Red', 'Blue'],   # color for each dataset
    alpha=0.4,               # transparancy for both groups
    label=['Female', 'Male'] # extra label for each datasets
)

ax.set_title('Age distributioin by Sex')
ax.set_xlabel('Age group')
ax.set_ylabel('Passenger #')
ax.xaxis.set_ticks([i for i in range(0, 90, 10)])
ax.xaxis.set_ticklabels(['{0}s'.format(i) for i in range(0, 90, 10)])

ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.spines['left'].set_visible(False)
ax.spines['bottom'].set_visible(True)
ax.legend()
