mpl.rcParams['font.size'] = 20.0    # globally assign fontsize for matplotlib text

fig = plt.figure(figsize=[16, 12])  # initialte figure object
ax = fig.add_subplot(1,1,1)         # add one axes in figure

ax.pie(
    y, # must-have, value to plot the pie chart
    labels=['Class: {0}'.format(i) for i in x], # specify the label outside the pie chart (usually) 
    explode=[0, 0, 0, 0, 0.1, 0],  # if any slice want to be emphasized
    autopct=lambda x: "{:7.1f}(%)".format(x),   # percent info, if needed inside the pie
    pctdistance=0.7,    # the location of percent info (away from center)
    labeldistance=1.05, # the location of label info (away from center)
    startangle=0,       # where the circle starts
    radius=1,           # the pie's radius
    counterclock=False, # clock wise or counter clock wise
    shadow=False)       # add extra shadow
ax.axis("equal")        # set axis as equal, so no distortion (get circle rather than oval)
ax.set_title('SibSp number distribution in a pie') # set title

plt.show()
